src/synqc/hardware/base.py

python
from __future__ import annotations

from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Dict, Any, Sequence


@dataclass
class HardwareProfile:
    """Describes a hardware family using simple scalar parameters.

    This is intentionally generic so we can support:
    - superconducting qubits
    - trapped-ion chains
    - neutral atoms
    - photonic systems

    without pulling in vendor-specific SDKs.
    """

    name: str
    params: Dict[str, float]


class HardwareBackend(ABC):
    """Abstract base class for SynQc hardware backends."""

    def __init__(self, profile: HardwareProfile) -> None:
        self.profile = profile

    @abstractmethod
    def run_cycle(
        self,
        drive1: Dict[str, float],
        probe: Dict[str, float],
        drive2: Dict[str, float],
    ) -> Dict[str, Any]:
        """Execute a single DPD cycle.

        Returns:
            A dict with keys such as:
            - "t": 1D array of time samples during probe
            - "y": 1D array of raw measurement samples (volts, counts, etc.)
            - "meta": arbitrary metadata (freq, power, etc.)
        """
        raise NotImplementedError

    @abstractmethod
    def true_params(self) -> Dict[str, float]:
        """Return the underlying 'true' plant parameters (for benchmarking)."""
        raise NotImplementedError

    def update_internal_drift(self) -> None:
        """Optional: update any internal drift between cycles."""
        # Default: no drift.
        return

    def run_sequence(
        self,
        sequence: Sequence[Dict[str, Dict[str, float]]],
    ) -> Sequence[Dict[str, Any]]:
        """Run a pre-built sequence of DPD cycles."""
        results = []
        for cycle in sequence:
            self.update_internal_drift()
            results.append(
                self.run_cycle(
                    drive1=cycle["drive1"],
                    probe=cycle["probe"],
                    drive2=cycle["drive2"],
                )
            )
        return results

src/synqc/hardware/synthetic.py

python
from __future__ import annotations

from dataclasses import dataclass
from typing import Dict, Any

import numpy as np

from .base import HardwareBackend, HardwareProfile


@dataclass
class SyntheticPlantParams:
    """Reference synthetic plant used for KPI tests.

    All frequencies are in rad/s, times in seconds.

    Attributes:
        delta: detuning  between drive and qubit resonance.
        g: coupling strength.
        T1: energy relaxation time.
        T2: dephasing time.
        snr_db: target SNR for probe measurements.
        jitter_std: timing jitter std dev during probe window (seconds).
    """

    delta: float  # rad/s
    g: float      # rad/s
    T1: float     # s
    T2: float     # s
    snr_db: float
    jitter_std: float


class SyntheticBackend(HardwareBackend):
    """Simple oscillator-like synthetic plant for closed-loop tests.

    We model a decaying sinusoid with detuning  and coupling g,
    observed with additive white Gaussian noise. This is not a full
    Bloch-equation integrator; it is a reproducible reference plant
    for KPIs and algorithm development.
    """

    def __init__(
        self,
        params: SyntheticPlantParams,
        seed: int | None = None,
    ) -> None:
        self.params = params
        self._rng = np.random.default_rng(seed)
        profile = HardwareProfile(
            name="synthetic",
            params={
                "delta": params.delta,
                "g": params.g,
                "T1": params.T1,
                "T2": params.T2,
            },
        )
        super().__init__(profile)

    def true_params(self) -> Dict[str, float]:
        p = self.params
        return {
            "delta": p.delta,
            "g": p.g,
            "T1": p.T1,
            "T2": p.T2,
        }

    def _generate_probe_signal(
        self,
        t_probe: float,
        dt: float,
        probe_freq: float,
        drive1_amp: float,
        drive2_amp: float,
    ) -> Dict[str, Any]:
        """Generate a noisy probe signal over the probe window.

        For now we use a decaying sinusoid with effective frequency
        (probe_freq + delta) and envelope based on T2.
        """
        n = max(int(t_probe / dt), 4)
        t = np.arange(n, dtype=float) * dt

        # Add simple timing jitter.
        if self.params.jitter_std > 0:
            jitter = self._rng.normal(scale=self.params.jitter_std, size=n)
            t_j = t + jitter
        else:
            t_j = t

        delta = self.params.delta
        g = self.params.g
        T2 = self.params.T2

        # Envelope and phase.
        envelope = np.exp(-t_j / max(T2, 1e-9))
        phase = (probe_freq + delta) * t_j

        # Simple dependence on drive amplitudes.
        amp = (abs(drive1_amp) + abs(drive2_amp)) * 0.5 * (g / (g + abs(delta) + 1e-9))

        signal = amp * envelope * np.cos(phase)

        # Add AWGN to achieve target SNR.
        snr_linear = 10.0 ** (self.params.snr_db / 10.0)
        power_signal = np.mean(signal**2) + 1e-12
        noise_power = power_signal / snr_linear
        noise_std = np.sqrt(noise_power)
        noise = self._rng.normal(scale=noise_std, size=n)

        y = signal + noise

        return {
            "t": t,
            "y": y,
            "meta": {
                "probe_freq": probe_freq,
                "delta_true": delta,
                "g_true": g,
                "T2": T2,
                "snr_db": self.params.snr_db,
                "noise_std": noise_std,
            },
        }

    def run_cycle(
        self,
        drive1: Dict[str, float],
        probe: Dict[str, float],
        drive2: Dict[str, float],
    ) -> Dict[str, Any]:
        t_probe = probe["duration"]
        dt = probe["dt"]
        probe_freq = probe["freq"]
        drive1_amp = drive1.get("amp", 1.0)
        drive2_amp = drive2.get("amp", 1.0)

        return self._generate_probe_signal(
            t_probe=t_probe,
            dt=dt,
            probe_freq=probe_freq,
            drive1_amp=drive1_amp,
            drive2_amp=drive2_amp,
        )

from __future__ import annotations

from dataclasses import dataclass
from typing import Dict, List


@dataclass
class DPDConfig:
    """Timing and control configuration for one DPD cycle.

    All times in seconds.
    """

    t_drive1: float
    t_probe: float
    t_drive2: float
    dt_probe: float
    probe_freq: float
    drive1_amp: float = 1.0
    drive2_amp: float = 1.0


@dataclass
class DPDCycle:
    """Concrete drive/probe/drive parameters for a single cycle."""

    drive1: Dict[str, float]
    probe: Dict[str, float]
    drive2: Dict[str, float]


class Scheduler:
    """Builds per-cycle DPD parameters and accumulates logs.

    This is a software scheduler only; it does not enforce real time.
    """

    def __init__(self) -> None:
        self._cycles: List[DPDCycle] = []

    def build_cycle(self, config: DPDConfig) -> DPDCycle:
        drive1 = {
            "duration": config.t_drive1,
            "amp": config.drive1_amp,
        }
        probe = {
            "duration": config.t_probe,
            "dt": config.dt_probe,
            "freq": config.probe_freq,
        }
        drive2 = {
            "duration": config.t_drive2,
            "amp": config.drive2_amp,
        }
        cycle = DPDCycle(drive1=drive1, probe=probe, drive2=drive2)
        self._cycles.append(cycle)
        return cycle

    @property
    def cycles(self) -> List[DPDCycle]:
        return list(self._cycles)

python
from __future__ import annotations

from dataclasses import dataclass
from typing import Dict

import numpy as np


@dataclass
class DemodResult:
    """Demodulation features extracted from a probe measurement."""

    I: float
    Q: float
    amp: float
    phase: float
    snr_est_db: float
    delta_meas: float


def demod_iq(
    t: np.ndarray,
    y: np.ndarray,
    ref_freq: float,
) -> DemodResult:
    """Simple IQ demodulation + rough delta estimate.

    Args:
        t: time samples (s)
        y: real-valued signal samples
        ref_freq: reference angular frequency _ref (rad/s)

    Returns:
        DemodResult with I/Q, amplitude, phase, SNR estimate and a crude
        detuning estimate delta_meas in rad/s.
    """
    if t.shape != y.shape:
        raise ValueError("t and y must have the same shape")

    # Reference carriers.
    cos_ref = np.cos(ref_freq * t)
    sin_ref = np.sin(ref_freq * t)

    I = float(2.0 * np.mean(y * cos_ref))
    Q = float(2.0 * np.mean(y * sin_ref))
    amp = float(np.hypot(I, Q))
    phase = float(np.arctan2(Q, I))

    # SNR estimate via signal and residual power.
    recon = 0.5 * amp * np.cos(ref_freq * t + phase)
    resid = y - recon
    ps = float(np.mean(recon**2) + 1e-12)
    pn = float(np.mean(resid**2) + 1e-12)
    snr_est_db = 10.0 * np.log10(ps / pn)

    # Crude delta estimate via phase slope over the window.
    # We fit phase(t)  phase0 + (ref_freq + delta) * t, so:
    # delta  d/dt - ref_freq.
    # Here we approximate d/dt by finite difference on the unwrapped phase.
    analytic = np.fft.ifft(np.fft.fft(y) * (2.0 * (t.size // 2) / t.size))
    # Fallback: if analytic fails (NaNs), just return 0 detuning.
    if np.any(~np.isfinite(analytic)):
        delta_meas = 0.0
    else:
        # Use Hilbert-style analytic signal for phase.
        # (This is a cheap approximation; we avoid importing scipy here.)
        # Construct a naive analytic signal:
        Y = np.fft.fft(y)
        h = np.zeros_like(Y)
        n = Y.size
        if n > 0:
            h[0] = 1.0
            if n % 2 == 0:
                h[1:n // 2] = 2.0
                h[n // 2] = 1.0
            else:
                h[1:(n + 1) // 2] = 2.0
        z = np.fft.ifft(Y * h)
        phi = np.unwrap(np.angle(z))
        dt = float(np.mean(np.diff(t))) if t.size > 1 else 1.0
        dphi_dt = float((phi[-1] - phi[0]) / max(dt * (t.size - 1), 1e-9))
        delta_meas = dphi_dt - ref_freq

    return DemodResult(
        I=I,
        Q=Q,
        amp=amp,
        phase=phase,
        snr_est_db=float(snr_est_db),
        delta_meas=float(delta_meas),
    )

(The analytic-signal bit is a bit hacky but keeps us in pure NumPy and gives us something to refine later.)

src/synqc/core/adapt.py

from __future__ import annotations

from dataclasses import dataclass


@dataclass
class AdaptState:
    """State of a 1D Kalman-style learner for detuning .

    x: current estimate of  (rad/s)
    P: current variance (rad/s)^2
    """

    x: float
    P: float


class Kalman1D:
    """Minimal scalar Kalman filter for tracking detuning.

    Model:
        x_k   = x_{k-1} + w_k,   w_k ~ N(0, Q)
        z_k   = x_k     + v_k,   v_k ~ N(0, R)

    For SynQc we interpret z_k as 'delta_meas' from demod_iq.
    """

    def __init__(self, x0: float, P0: float, Q: float, R: float) -> None:
        if P0 <= 0 or Q < 0 or R <= 0:
            raise ValueError("P0 > 0, Q >= 0, R > 0 required")
        self.state = AdaptState(x=float(x0), P=float(P0))
        self.Q = float(Q)
        self.R = float(R)

    def predict(self) -> None:
        """Time update."""
        self.state.P += self.Q

    def update(self, z: float) -> AdaptState:
        """Measurement update.

        Args:
            z: new measurement of detuning (rad/s)

        Returns:
            Updated AdaptState.
        """
        # Kalman gain.
        S = self.state.P + self.R
        K = self.state.P / S

        # Innovation and state update.
        y = z - self.state.x
        self.state.x += K * y

        # Covariance update (Joseph form not needed in 1D).
        self.state.P = (1.0 - K) * self.state.P

        return self.state

    def step(self, z: float) -> AdaptState:
        self.predict()
        return self.update(z)

6. Example: closed-loop synthetic DPD

examples/run_synthetic_closed_loop.py

python
from __future__ import annotations

import math

import numpy as np

from synqc import (
    DPDConfig,
    Scheduler,
    SyntheticBackend,
    SyntheticPlantParams,
    demod_iq,
    Kalman1D,
)


def main() -> None:
    # Reference plant parameters (roughly aligned with your spec ranges).
    params = SyntheticPlantParams(
        delta=2.0 * math.pi * 0.5e6,   # 0.5 MHz detuning
        g=2.0 * math.pi * 5.0e6,       # 5 MHz coupling
        T1=30e-6,
        T2=40e-6,
        snr_db=30.0,
        jitter_std=50e-9,
    )
    backend = SyntheticBackend(params, seed=1234)

    # DPD timing configuration (probe-only model).
    scheduler = Scheduler()
    cfg = DPDConfig(
        t_drive1=40e-9,
        t_probe=200e-9,
        t_drive2=40e-9,
        dt_probe=1e-9,
        probe_freq=2.0 * math.pi * 5.0e9,
        drive1_amp=1.0,
        drive2_amp=1.0,
    )

    # Kalman filter initialisation: we pretend we are quite uncertain about .
    kf = Kalman1D(
        x0=0.0,                     # initial  estimate
        P0=(2.0 * math.pi * 2e6) ** 2,  # very uncertain (2 MHz)
        Q=(2.0 * math.pi * 10e3) ** 2,  # small process noise (drift)
        R=(2.0 * math.pi * 0.2e6) ** 2,  # measurement noise
    )

    n_cycles = 300
    estimates = []
    true_delta = backend.true_params()["delta"]

    for k in range(n_cycles):
        cycle = scheduler.build_cycle(cfg)
        raw = backend.run_cycle(
            drive1=cycle.drive1,
            probe=cycle.probe,
            drive2=cycle.drive2,
        )

        t = raw["t"]
        y = raw["y"]
        ref_freq = cfg.probe_freq

        demod = demod_iq(t=t, y=y, ref_freq=ref_freq)
        state = kf.step(demod.delta_meas)
        estimates.append(state.x)

        if (k + 1) % 50 == 0:
            err = (state.x - true_delta) / true_delta * 100.0
            print(
                f"Cycle {k+1:3d}: _est = {state.x/2/math.pi/1e6:7.3f} MHz "
                f"(true {true_delta/2/math.pi/1e6:7.3f} MHz) "
                f"err = {err:7.3f}%  | SNR_est  {demod.snr_est_db:5.1f} dB"
            )

    final_est = estimates[-1]
    rel_err = abs(final_est - true_delta) / abs(true_delta)
    print("\n=== Summary ===")
    print(
        f"True  = {true_delta/2/math.pi/1e6:.3f} MHz, "
        f"Estimated  = {final_est/2/math.pi/1e6:.3f} MHz, "
        f"Relative error = {rel_err*100:.2f}%"
    )
    if rel_err <= 0.02:
        print("KPI:  recovered within 2%  PASS")
    else:
        print("KPI:  recovered within 2%  FAIL")


if __name__ == "__main__":
    main()

This is the hello world of SynQc: it runs the whole DPD loop against a synthetic plant, uses a Kalman-style learner, and checks the 2% detuning recovery KPI from your readiness doc.

# SynQc: Temporal Dynamics Series

**Goal:** Give experimenters a small, cross-platform Python library that
implements the core *DriveProbeDrive (DPD)* loop for quantum control,
with a clean path to plug in real hardware backends.

This v0.3-alpha release includes:

- a **scheduler** that builds DPD cycles (drive  probe  drive),
- a **synthetic hardware backend** that behaves like a noisy,
  decaying oscillator with detuning , coupling g, and decoherence
  times T/T,
- a simple **IQ demodulation** pipeline,
- a 1D **Kalman-style learner** that tracks detuning  over many cycles,
- a **closed-loop example** that runs the whole system and checks
  whether we recover  within 2% on the reference plant.

Everything is pure Python + NumPy and runs on Windows, macOS, and Linux.

Plugging in real hardware

To connect your own device:

  1. Subclass synqc.hardware.base.HardwareBackend.

  2. Implement run_cycle(drive1, probe, drive2) to:

    • emit a driveprobedrive sequence on your control stack,

    • collect a stream of probe measurements during the probe window,

    • return {"t": t_array, "y": y_array, "meta": {...}}.

  3. Use Scheduler, demod_iq, and Kalman1D as shown in
    examples/run_synthetic_closed_loop.py, swapping in your backend.

We will extend this base with:

  • mid-circuit measurement support,

  • multi-parameter adaptation (, g, T, T, ...),

  • hardware profiles for superconducting, trapped-ion, neutral-atom,
    and photonic platforms,

in future versions, staying aligned with the formal spec and KPI
documents in this repository.
